local function playAnim(isLeft, dur)

local function findTarget()
    local nearest, dist = nil, RANGE
    for _, obj in pairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj:FindFirstChild("HumanoidRootPart") then
            if obj ~= char and obj.Humanoid.Health > 0 then
                local d = (obj.HumanoidRootPart.Position - root.Position).Magnitude
                if d < dist then nearest = obj dist = d end
            end
        end
    end
    return nearest
end

local function executeDash()
    if dashing or not root then return end
    local target = (selectedPlr and selectedPlr.Character) or findTarget()
    if not target or not target:FindFirstChild("HumanoidRootPart") then return end

    dashing = true
    local tRoot = target.HumanoidRootPart
    local tPos, myPos = tRoot.Position, root.Position
    local tLook = tRoot.CFrame.LookVector
    local dot = tLook:Dot(myPos - tPos)
    local endPos = (dot > 0) and (tPos - tLook * F_OFFSET) or (tPos + tLook * B_OFFSET)
    endPos = Vector3.new(endPos.X, tPos.Y + 0.5, endPos.Z)
    
    local horDist = (Vector3.new(myPos.X, 0, myPos.Z) - Vector3.new(tPos.X, 0, tPos.Z)).Magnitude
    local clampDist = math.clamp(studs, MIN_DIST, MAX_DIST)
    local dashOff = (Vector3.new(endPos.X, 0, endPos.Z) - Vector3.new(tPos.X, 0, tPos.Z)).Magnitude
    local angleTo = math.atan2(myPos.Z - tPos.Z, myPos.X - tPos.X)
    local targetAngle = math.atan2(endPos.Z - tPos.Z, endPos.X - tPos.X)
    local angleOff = targetAngle - angleTo
    while math.pi < angleOff do angleOff = angleOff - 2 * math.pi end
    while angleOff < -math.pi do angleOff = angleOff + 2 * math.pi end
    
    local cd = BASE_CD / math.max(speedMult, 0.05)
    playAnim(angleOff > 0, cd)
    
    local cam = Workspace.CurrentCamera
    local autoRot = hum.AutoRotate
    hum.AutoRotate = false
    
    if dashKeyOn then fireEvent({[1] = {Dash = Enum.KeyCode.W, Key = Enum.KeyCode.Q, Goal = "KeyPress"}}) end
    if m1On then task.delay(0.1, function() fireEvent({[1] = {Mobile = true, Goal = "LeftClick"}}) task.wait(0.05) fireEvent({[1] = {Goal = "LeftClickRelease", Mobile = true}}) end) end

    local startTime, lastUp = tick(), tick()
    local conn
    conn = RunService.Heartbeat:Connect(function()
        local now = tick()
        local prog = math.clamp((now - startTime) / cd, 0, 1)
        if prog >= 1 or not tRoot or not tRoot.Parent or not root then
            conn:Disconnect()
            root.AssemblyLinearVelocity = Vector3.zero
            hum.AutoRotate = autoRot
            task.wait(0.1)
            dashing = false
            return
        end
        local eased = 1 - (1 - prog) ^ 3
        local curDist = (prog <= 0.5) and (horDist + (clampDist - horDist) * (1 - (1 - (prog/0.5))^3)) or (clampDist + (dashOff - clampDist) * (1 - (1 - ((prog-0.5)/0.5))^3))
        local curTargetAng = angleTo + angleOff * eased
        local newPos = Vector3.new(tPos.X + curDist * math.cos(curTargetAng), myPos.Y + (endPos.Y - myPos.Y) * eased, tPos.Z + curDist * math.sin(curTargetAng))
        root.AssemblyLinearVelocity = (newPos - root.Position) / math.max(0.0001, now - lastUp)
        root.CFrame = CFrame.new(root.Position, Vector3.new(tPos.X, root.Position.Y, tPos.Z))
        cam.CFrame = CFrame.new(cam.CFrame.Position, tRoot.Position)
        lastUp = now
    end)
end

local function getClosestPlayer()
    local closest, shortest = nil, maxDistance
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character then
            local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
            local hum = plr.Character:FindFirstChild("Humanoid")
            if hrp and hum and hum.Health > 0 then
                local dist = (hrp.Position - Camera.CFrame.Position).Magnitude
                if dist < shortest then
                    shortest = dist
                    closest = hrp
                end
            end
        end
    end
    return closest
end

local function getDummyTarget()
    local dummy = workspace:FindFirstChild("Live") and workspace.Live:FindFirstChild("Weakest Dummy")
    if dummy then
        local humanoid = dummy:FindFirstChild("Humanoid")
        local torso = dummy:FindFirstChild("Torso")
        if humanoid and humanoid.Health > 0 and torso then
            return torso
        end
    end
    return nil
end

local function getClosestPlayer()
    local closest = nil
    local shortest = maxDistance
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character then
            local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
            local hum = plr.Character:FindFirstChild("Humanoid")
            if hrp and hum and hum.Health > 0 then
                local dist = (hrp.Position - Camera.CFrame.Position).Magnitude
                if dist < shortest then
                    shortest = dist
                    closest = hrp
                end
            end
        end
    end
    return closest
end

local function GetVictimModel(hitAttr)
    if not hitAttr then return nil end
    local name = hitAttr:match("([^;]+)")

        local dummy = live:FindFirstChild(name)
        if dummy then return dummy end

                if descendant:IsA("ParticleEmitter") or descendant:IsA("Trail") then
                    descendant:Destroy()

                m.Part1 = c
                m.Parent = m.Part0
                table.insert(ActiveItems, m)

            table.insert(ActiveItems, c)
        end
    end
    
    setupDeathPart({folder.HandleL}, true)
    setupDeathPart({folder.HandleR}, false)
    setupDeathPart({folder:FindFirstChild("Left"), folder:FindFirstChild("Right")}, false, char.PrimaryPart)
end
